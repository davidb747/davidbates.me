```shell
# Undo commits not already pushed to source
git reset --hard HEAD~1

# Reset to upstream branch
git reset --hard origin/mybranch

# Remove all untracked files from repository
git clean -xdf

# Abort merge
git merge --abort

# Abort cherry pick
git cherry-pick --abort
```
